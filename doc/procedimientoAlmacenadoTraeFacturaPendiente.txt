ALTER   procedure [dbo].[traeXml]
as
begin
select	d.IDTIPOORDEN,
		d.IDORDEN,
		d.idCliente,
		d.nombreTercero,
		convert(varchar(10),d.fechaDcto,111) fechaDcto,
		f.iddcto, 
		f.idDctoDian,
		f.prefijoDian,
		f.status,
		f.url
from tblDctos d 
inner join tblFacturaElectronica f
on f.idOrden = d.idOrden
and f.idtipoorden = d.IDTIPOORDEN
where f.tipo_xml = 1
and f.status = 'false'
end
return
