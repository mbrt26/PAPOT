      ,[referenciaCliente]
      ,tmpFIC.idOperacion
      ,[idPlu]
      ,[nombreReferencia]

      ,[referencia]

(SELECT [idEscala]
      ,[idOperacion]
      ,[estado]
      ,[idOrdenSalida]

      ,[idFicha]
